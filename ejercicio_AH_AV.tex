\subsection{Instrucciones}\label{instrucciones}

\begin{itemize}
\tightlist
\item
  \textbf{Objetivo}: calcular \textbf{Δ (absoluta)}, \textbf{Δ\%} y
  \textbf{AV} (t-1 y t) para EERR y ESF; luego \textbf{analizar}.
\item
  \textbf{Base AV}: en \textbf{EERR}, ( \text{Ventas Netas} = 100\% ).
  En \textbf{ESF}, ( \text{Total Activos} = 100\% ).
\item
  \textbf{Formato}: complete a mano en las líneas. Use \textbf{MM\$}
  (millones de pesos). Redondee \textbf{AV} a \textbf{1 decimal} y
  \textbf{Δ\%} a \textbf{1 decimal}.
\end{itemize}

\subsubsection{Fórmulas de referencia}\label{fuxf3rmulas-de-referencia}

\[
\Delta_{\text{abs}} = V_t - V_{t-1}
\qquad
\Delta\% = \frac{V_t - V_{t-1}}{\lvert V_{t-1} \rvert}\times 100
\qquad
AV_i = \frac{\text{Cuenta } i}{\text{Base}} \times 100
\]

\begin{quote}
Si (V\_\{t-1\}=0) o cambia de signo, marque como \textbf{no comparable
(N/C)} y explique.
\end{quote}

\subsubsection{Checklist de consistencia
rápida}\label{checklist-de-consistencia-ruxe1pida}

\begin{itemize}
\tightlist
\item
  ( \text{Ventas} - \text{COGS} = \text{G. Bruta} ); ( \text{G. Bruta} -
  \text{OPEX} = \text{Op.} ).\\
\item
  En EERR, ( \sum AV = 100\% ) (base ventas) por periodo.\\
\item
  En ESF, ( \sum AV = 100\% ) (base total activos) por periodo.\\
\item
  ( t-1 + \Delta\_\{\text{abs}\} = t ) en cada fila.\\
\item
  Señale \textbf{N/C} cuando corresponda y explique.
\end{itemize}
